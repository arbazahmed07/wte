<script>

let tasks = [];

function addTask() {

    let task =
        document.getElementById("task").value;

    let priority =
        document.getElementById("priority").value;

    if (task == "") {
        alert("Enter task");
        return;
    }

    tasks.push({
        task: task,
        priority: priority,
        completed: false
    });

    document.getElementById("task").value = "";

    displayTasks();
}

function displayTasks() {

    let filter =
        document.getElementById("filter").value;

    let table =
        document.getElementById("table");

    table.innerHTML = `
        <tr>
            <th>Task</th>
            <th>Priority</th>
            <th>Status</th>
        </tr>
    `;

    let completed = 0;
    let pending = 0;

    for (let i = 0; i < tasks.length; i++) {

        if (filter == "All" ||
            filter == tasks[i].priority) {

            let status =
                tasks[i].completed ?
                "Completed" : "Pending";

            let row = `
                <tr>
                    <td>${tasks[i].task}</td>
                    <td>${tasks[i].priority}</td>
                    <td>
                        <input type="checkbox"
                        ${tasks[i].completed ? "checked" : ""}
                        onchange="toggleTask(${i})">

                        ${status}
                    </td>
                </tr>
            `;

            table.innerHTML += row;
        }

        if (tasks[i].completed) {
            completed++;
        }
        else {
            pending++;
        }
    }

    document.getElementById("completedCount").innerHTML =
        completed;

    document.getElementById("pendingCount").innerHTML =
        pending;
}

function toggleTask(index) {

    tasks[index].completed =
        !tasks[index].completed;

    displayTasks();
}

</script>
