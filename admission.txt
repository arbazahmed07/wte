<script>

function calculate() {

    let physics =
        Number(document.getElementById("physics").value);

    let chemistry =
        Number(document.getElementById("chemistry").value);

    let maths =
        Number(document.getElementById("maths").value);

    let entrance =
        Number(document.getElementById("entrance").value);

    let total = physics + chemistry + maths;

    let percentage = total / 3;

    document.getElementById("total").innerHTML = total;

    document.getElementById("percentage").innerHTML =
        percentage.toFixed(2);

    document.getElementById("percentageValue").value =
        percentage.toFixed(2);

    if (percentage >= 70 && entrance >= 60) {
        document.getElementById("preview").innerHTML =
            "Eligible";
    }
    else {
        document.getElementById("preview").innerHTML =
            "Not Eligible";
    }
}

function validateForm() {

    let physics =
        Number(document.getElementById("physics").value);

    let chemistry =
        Number(document.getElementById("chemistry").value);

    let maths =
        Number(document.getElementById("maths").value);

    if (physics < 0 || physics > 100 ||
        chemistry < 0 || chemistry > 100 ||
        maths < 0 || maths > 100) {

        alert("Marks must be between 0 and 100");

        return false;
    }

    return true;
}

//AdmissionServlet.java
import java.io.*;
import javax.servlet.*;
import javax.servlet.http.*;

public class AdmissionServlet extends HttpServlet {

    protected void doPost(HttpServletRequest request,
                          HttpServletResponse response)
                          throws ServletException, IOException {

        String name = request.getParameter("name");

        int physics =
            Integer.parseInt(request.getParameter("physics"));

        int chemistry =
            Integer.parseInt(request.getParameter("chemistry"));

        int maths =
            Integer.parseInt(request.getParameter("maths"));

        int entrance =
            Integer.parseInt(request.getParameter("entrance"));

        double percentage =
            Double.parseDouble(request.getParameter("percentage"));

        String result;

        if (percentage >= 70 && entrance >= 60) {
            result = "Eligible for Admission";
        }
        else {
            result = "Not Eligible for Admission";
        }

        request.setAttribute("name", name);
        request.setAttribute("physics", physics);
        request.setAttribute("chemistry", chemistry);
        request.setAttribute("maths", maths);
        request.setAttribute("entrance", entrance);
        request.setAttribute("percentage", percentage);
        request.setAttribute("result", result);

        RequestDispatcher rd =
            request.getRequestDispatcher("result.jsp");

        rd.forward(request, response);
    }
}
