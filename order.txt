function calculate() {

    let burger = document.getElementById("burger").value * 100;
    let pizza = document.getElementById("pizza").value * 200;
    let sandwich = document.getElementById("sandwich").value * 80;
    let pasta = document.getElementById("pasta").value * 150;
    let fries = document.getElementById("fries").value * 60;
    let coke = document.getElementById("coke").value * 40;

    document.getElementById("burgerTotal").innerHTML = burger;
    document.getElementById("pizzaTotal").innerHTML = pizza;
    document.getElementById("sandwichTotal").innerHTML = sandwich;
    document.getElementById("pastaTotal").innerHTML = pasta;
    document.getElementById("friesTotal").innerHTML = fries;
    document.getElementById("cokeTotal").innerHTML = coke;

    let total = burger + pizza + sandwich + pasta + fries + coke;

    document.getElementById("grandTotal").innerHTML = total;

    let delivery = 0;

    if (total < 500 && total > 0) {
        delivery = 50;
    }

    document.getElementById("delivery").innerHTML = delivery;

    let finalTotal = total + delivery;

    document.getElementById("finalTotal").innerHTML = finalTotal;

    document.getElementById("finalAmount").value = finalTotal;
}

</script>

import java.io.*;
import jakarta.servlet.*;
import jakarta.servlet.http.*;

public class OrderServlet extends HttpServlet {

    protected void doPost(HttpServletRequest request,
                          HttpServletResponse response)
                          throws ServletException, IOException {

        request.setAttribute("burgerQty", request.getParameter("burgerQty"));
        request.setAttribute("pizzaQty", request.getParameter("pizzaQty"));
        request.setAttribute("sandwichQty", request.getParameter("sandwichQty"));
        request.setAttribute("pastaQty", request.getParameter("pastaQty"));
        request.setAttribute("friesQty", request.getParameter("friesQty"));
        request.setAttribute("cokeQty", request.getParameter("cokeQty"));

        request.setAttribute("finalAmount",
                request.getParameter("finalAmount"));

        RequestDispatcher rd =
                request.getRequestDispatcher("result.jsp");

        rd.forward(request, response);
    }
}
