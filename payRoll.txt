function validateForm() {

    let name =
        document.getElementById("name").value;

    let basic =
        document.getElementById("basic").value;

    let hra =
        document.getElementById("hra").value;

    let bonus =
        document.getElementById("bonus").value;

    if (name == "" ||
        basic == "" ||
        hra == "" ||
        bonus == "") {

        alert("Fill all fields");

        return false;
    }

    if (basic <= 0 ||
        hra < 0 ||
        bonus < 0) {

        alert("Enter valid values");

        return false;
    }

    return true;
}

//PayrollServlet.java

import java.io.*;
import javax.servlet.*;
import javax.servlet.http.*;

public class PayrollServlet extends HttpServlet {

    protected void doPost(HttpServletRequest request,
                          HttpServletResponse response)
                          throws ServletException, IOException {

        String name =
            request.getParameter("name");

        double basic =
            Double.parseDouble(request.getParameter("basic"));

        double hraPercent =
            Double.parseDouble(request.getParameter("hra"));

        double bonus =
            Double.parseDouble(request.getParameter("bonus"));

        double hra =
            (basic * hraPercent) / 100;

        double gross =
            basic + hra + bonus;

        request.setAttribute("name", name);
        request.setAttribute("basic", basic);
        request.setAttribute("hra", hra);
        request.setAttribute("bonus", bonus);
        request.setAttribute("gross", gross);

        RequestDispatcher rd =
            request.getRequestDispatcher("result.jsp");

        rd.forward(request, response);
    }
}
